package repositories

import (
	"context"
	"example/rest-api-demo/src/models"
)

type CommentsRepository interface {
	CreateComment(c models.Comment) (string, error)
	GetComments(page, size int, ctx context.Context) ([]*models.Comment, error)
	GetCommentByID(id string, ctx context.Context) (*models.Comment, error)
	DeleteCommentByID(id string) error
	UpdateCommentByID(id string, c models.Comment) error
}


type MoviesRepository interface {
	GetMovies(page, size int, ctx context.Context) ([]*models.Movie, error)
	GetMovieByID(objID primitive.ObjectID, ctx context.Context) (*models.Movie, error)
}
